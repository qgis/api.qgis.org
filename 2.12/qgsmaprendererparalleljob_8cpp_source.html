49 void QgsMapRendererParallelJob::start()
50 {

53 
54  mRenderingStart.start();
55 
56  mStatus = RenderingLayers;
57 
58  delete mLabelingEngine;
59  mLabelingEngine = 0;
60 
61  delete mLabelingEngineV2;
62  mLabelingEngineV2 = 0;
63 
64  if ( mSettings.testFlag( QgsMapSettings::DrawLabeling ) )

77  mLayerJobs = prepareJobs( 0, mLabelingEngine, mLabelingEngineV2 );

89 void QgsMapRendererParallelJob::cancel()
90 {
91  if ( !isActive() )
92  return;
93 
94  QgsDebugMsg( QString( "PARALLEL cancel at status %1" ).arg( mStatus ) );
95 
96  mLabelingRenderContext.setRenderingStopped( true );
97  for ( LayerRenderJobs::iterator it = mLayerJobs.begin(); it != mLayerJobs.end(); ++it )
98  {
99  it->context.setRenderingStopped( true );
100  }
101 
102  if ( mStatus == RenderingLayers )
103  {
104  disconnect( &mFutureWatcher, SIGNAL( finished() ), this, SLOT( renderLayersFinished() ) );
105 
106  mFutureWatcher.waitForFinished();
107 
108  renderLayersFinished();
109  }
110 
111  if ( mStatus == RenderingLabels )
112  {
113  disconnect( &mLabelingFutureWatcher, SIGNAL( finished() ), this, SLOT( renderingFinished() ) );
114 
115  mLabelingFutureWatcher.waitForFinished();
116 
117  renderingFinished();
118  }
119 
120  Q_ASSERT( mStatus == Idle );
121 }

182 void QgsMapRendererParallelJob::renderLayersFinished()
183 {
184  Q_ASSERT( mStatus == RenderingLayers );
185 
186  // compose final image
187  mFinalImage = composeImage( mSettings, mLayerJobs );
188 
189  cleanupJobs( mLayerJobs );
190 
191  QgsDebugMsg( "PARALLEL layers finished" );
192 
193  if ( mSettings.testFlag( QgsMapSettings::DrawLabeling ) && !mLabelingRenderContext.renderingStopped() )
194  {
195  mStatus = RenderingLabels;
196 
197  connect( &mLabelingFutureWatcher, SIGNAL( finished() ), this, SLOT( renderingFinished() ) );
198 
199  // now start rendering of labeling!
200  mLabelingFuture = QtConcurrent::run( renderLabelsStatic, this );
201  mLabelingFutureWatcher.setFuture( mLabelingFuture );
202  }
203  else
204  {
205  renderingFinished();
206  }
207 }

220 void QgsMapRendererParallelJob::renderLayerStatic( LayerRenderJob& job )
221 {
222  if ( job.context.renderingStopped() )
223  return;
224 
225  if ( job.cached )
226  return;
227 
228  QTime t;
229  t.start();
230  QgsDebugMsg( QString( "job %1 start" ).arg(( ulong ) &job, 0, 16 ) );
231 
232  try
233  {
234  job.renderer->render();
235  }
236  catch ( QgsException & e )
237  {
238  QgsDebugMsg( "Caught unhandled QgsException: " + e.what() );
239  }
240  catch ( std::exception & e )
241  {
242  QgsDebugMsg( "Caught unhandled std::exception: " + QString::fromAscii( e.what() ) );
243  }
244  catch ( ... )
245  {
246  QgsDebugMsg( "Caught unhandled unknown exception" );
247  }
248 
249  int tt = t.elapsed();
250  QgsDebugMsg( QString( "job %1 end [%2 ms]" ).arg(( ulong ) &job, 0, 16 ).arg( tt ) );
251  Q_UNUSED( tt );
252 }
