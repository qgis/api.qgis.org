74

83  {

85  for ( int j = 0; j < mNumColumns; ++j )
86  {
87  if ( mInterpolator->interpolatePoint( currentXValue, currentYValue, interpolatedValue ) == 0 )
88  {

90  }

93  outStream << "-9999 ";
94  }

96  }

104  outputFile.remove();
105  return 3;

109  }
