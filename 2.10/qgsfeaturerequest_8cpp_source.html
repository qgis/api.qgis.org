130 QgsFeatureRequest& QgsFeatureRequest::setSubsetOfAttributes( const QStringList& attrNames, const QgsFields& fields )
131 {
132  if ( attrNames.contains( QgsFeatureRequest::AllAttributes ) )
133  {
134  //attribute string list contains the all attributes flag, so we must fetch all attributes
135  return *this;

157 bool QgsFeatureRequest::acceptFeature( const QgsFeature& feature )
158 {
159  switch ( mFilter )
160  {
161  case QgsFeatureRequest::FilterNone:
162  return true;
163  break;
164 
165  case QgsFeatureRequest::FilterRect:
166  if ( feature.constGeometry() && feature.constGeometry()->intersects( mFilterRect ) )
167  return true;
168  else
169  return false;
170  break;
171 
172  case QgsFeatureRequest::FilterFid:
173  if ( feature.id() == mFilterFid )
174  return true;
175  else
176  return false;
177  break;
178 
179  case QgsFeatureRequest::FilterExpression:
180  if ( mFilterExpression->evaluate( feature ).toBool() )
181  return true;
182  else
183  return false;
184  break;
185 
186  case QgsFeatureRequest::FilterFids:
187  if ( mFilterFids.contains( feature.id() ) )
188  return true;
189  else
190  return false;
191  break;
192  }
193 
194  return true;
195 }
