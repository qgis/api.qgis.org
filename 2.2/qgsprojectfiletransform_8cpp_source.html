724  }
725  }

727  {

729  QString type = customColorRampElem.firstChildElement( "colorRampType" ).text();

738  {
739  currentItemElem = colorNodeList.at( i ).toElement();
740  value = currentItemElem.attribute( "value" );
741  label = currentItemElem.attribute( "label" );
742  red = currentItemElem.attribute( "red" ).toInt();
743  green = currentItemElem.attribute( "green" ).toInt();
744  blue = currentItemElem.attribute( "blue" ).toInt();

746  QDomElement newItemElem = doc.createElement( "item" );
747  newItemElem.setAttribute( "value", value );
748  newItemElem.setAttribute( "label", label );
749  newItemElem.setAttribute( "color", newColor.name() );
750  newColorRampShaderElem.appendChild( newItemElem );
751  }
752  }
753  }
754  else if ( drawingStyle == "PalettedColor" )
755  {
756  rasterRendererElem.setAttribute( "type", "paletted" );
757  rasterRendererElem.setAttribute( "band", grayBand );
758  QDomElement customColorRampElem = rasterPropertiesElem.firstChildElement( "customColorRamp" );
759  QDomNodeList colorRampEntryList = customColorRampElem.elementsByTagName( "colorRampEntry" );
760  QDomElement newColorPaletteElem = doc.createElement( "colorPalette" );
761 
762  int red = 0;
763  int green = 0;
764  int blue = 0;
765  int value = 0;
766  QDomElement colorRampEntryElem;
767  for ( int i = 0; i < colorRampEntryList.size(); ++i )
768  {
769  colorRampEntryElem = colorRampEntryList.at( i ).toElement();
770  QDomElement newPaletteElem = doc.createElement( "paletteEntry" );
771  value = ( int )( colorRampEntryElem.attribute( "value" ).toDouble() );
772  newPaletteElem.setAttribute( "value", value );
773  red = colorRampEntryElem.attribute( "red" ).toInt();
774  green = colorRampEntryElem.attribute( "green" ).toInt();
775  blue = colorRampEntryElem.attribute( "blue" ).toInt();
776  newPaletteElem.setAttribute( "color", QColor( red, green, blue ).name() );
777  QString label = colorRampEntryElem.attribute( "label" );
778  if ( !label.isEmpty() )
779  {
780  newPaletteElem.setAttribute( "label", label );
781  }
782  newColorPaletteElem.appendChild( newPaletteElem );
783  }
784  rasterRendererElem.appendChild( newColorPaletteElem );
785  }
786  else if ( drawingStyle == "MultiBandColor" )
787  {
788  rasterRendererElem.setAttribute( "type", "multibandcolor" );

803  }

834 void QgsProjectFileTransform::transformContrastEnhancement( QDomDocument& doc, const QDomElement& rasterproperties, QDomElement& rendererElem )
835 {
836  if ( rasterproperties.isNull() || rendererElem.isNull() )
837  {
838  return;
839  }
840 
841  double minimumValue = 0;
842  double maximumValue = 0;
843  QDomElement contrastMinMaxElem = rasterproperties.firstChildElement( "contrastEnhancementMinMaxValues" );
844  if ( contrastMinMaxElem.isNull() )
845  {
846  return;
847  }
848 
849  QDomElement contrastEnhancementAlgorithmElem = rasterproperties.firstChildElement( "mContrastEnhancementAlgorithm" );
850  if ( contrastEnhancementAlgorithmElem.isNull() )
851  {
852  return;
853  }
854 
855  //convert enhancement name to enumeration
856  int algorithmEnum = 0;
857  QString algorithmString = contrastEnhancementAlgorithmElem.text();
858  if ( algorithmString == "StretchToMinimumMaximum" )
859  {
860  algorithmEnum = 1;
861  }
862  else if ( algorithmString == "StretchAndClipToMinimumMaximum" )
863  {
864  algorithmEnum = 2;
865  }
866  else if ( algorithmString == "ClipToMinimumMaximum" )
867  {
868  algorithmEnum = 3;
869  }
870  else if ( algorithmString == "UserDefinedEnhancement" )
871  {
872  algorithmEnum = 4;
873  }
874 
875  QDomNodeList minMaxEntryList = contrastMinMaxElem.elementsByTagName( "minMaxEntry" );
876  QStringList enhancementNameList;
877  if ( minMaxEntryList.size() == 1 )
878  {
879  enhancementNameList << "contrastEnhancement";
880  }
881  if ( minMaxEntryList.size() == 3 )
882  {
883  enhancementNameList << "redContrastEnhancement" << "greenContrastEnhancement" << "blueContrastEnhancement";
884  }
885  if ( minMaxEntryList.size() > enhancementNameList.size() )
886  {
887  return;
888  }
889 
890  QDomElement minMaxEntryElem;
891  for ( int i = 0; i < minMaxEntryList.size(); ++i )
892  {
893  minMaxEntryElem = minMaxEntryList.at( i ).toElement();
894  QDomElement minElem = minMaxEntryElem.firstChildElement( "min" );
895  if ( minElem.isNull() )
896  {
897  return;
898  }
899  minimumValue = minElem.text().toDouble();
900 
901  QDomElement maxElem = minMaxEntryElem.firstChildElement( "max" );
902  if ( maxElem.isNull() )
903  {
904  return;
905  }
906  maximumValue = maxElem.text().toDouble();
907 
908  QDomElement newContrastEnhancementElem = doc.createElement( enhancementNameList.at( i ) );
909  QDomElement newMinValElem = doc.createElement( "minValue" );
910  QDomText minText = doc.createTextNode( QString::number( minimumValue ) );
911  newMinValElem.appendChild( minText );
912  newContrastEnhancementElem.appendChild( newMinValElem );
913  QDomElement newMaxValElem = doc.createElement( "maxValue" );
914  QDomText maxText = doc.createTextNode( QString::number( maximumValue ) );
915  newMaxValElem.appendChild( maxText );
916  newContrastEnhancementElem.appendChild( newMaxValElem );
917 
918  QDomElement newAlgorithmElem = doc.createElement( "algorithm" );
919  QDomText newAlgorithmText = doc.createTextNode( QString::number( algorithmEnum ) );
920  newAlgorithmElem.appendChild( newAlgorithmText );
921  newContrastEnhancementElem.appendChild( newAlgorithmElem );
922 
923  rendererElem.appendChild( newContrastEnhancementElem );
924  }
