81

98  }

102  }

104  {

111  str += "<p><b>" + m.tag() + ":</b> " + m.message();
112 #ifdef QGISDEBUG

114  if ( !srcUrl.isEmpty() )
115  {
116  QString url = QString( "%1/%2#L%3" ).arg( srcUrl, file ).arg( m.line() );
117  str += QString( "<br>(<a href='%1'>%2</a>)" ).arg( url, location );
118  }

122  }
123 #endif
124  }
