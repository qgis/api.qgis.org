93 void QgsSymbolButton::showSettingsDialog()
94 {
95  QgsExpressionContext context;
96  if ( mExpressionContextGenerator )
97  context = mExpressionContextGenerator->createExpressionContext();
98  else
99  {
100  context.appendScopes( QgsExpressionContextUtils::globalProjectLayerScopes( mLayer.data() ) );
101  }
102  QgsSymbol *newSymbol = mSymbol->clone();
103 
104  QgsSymbolWidgetContext symbolContext;
105  symbolContext.setExpressionContext( &context );
106  symbolContext.setMapCanvas( mMapCanvas );
107  symbolContext.setMessageBar( mMessageBar );
108 
109  QgsPanelWidget *panel = QgsPanelWidget::findParentPanel( this );
110  if ( panel && panel->dockMode() )
111  {
112  QgsSymbolSelectorWidget *d = new QgsSymbolSelectorWidget( newSymbol, QgsStyle::defaultStyle(), mLayer, nullptr );
113  d->setPanelTitle( mDialogTitle );
114  d->setContext( symbolContext );
115  connect( d, &QgsPanelWidget::widgetChanged, this, &QgsSymbolButton::updateSymbolFromWidget );
116  connect( d, &QgsPanelWidget::panelAccepted, this, &QgsSymbolButton::cleanUpSymbolSelector );
117  panel->openPanel( d );
118  }
119  else
120  {
121  QgsSymbolSelectorDialog dialog( newSymbol, QgsStyle::defaultStyle(), mLayer, nullptr );
122  dialog.setWindowTitle( mDialogTitle );
123  dialog.setContext( symbolContext );
124  if ( dialog.exec() )
125  {
126  setSymbol( newSymbol );
127  }
128  else
129  {
130  delete newSymbol;
131  }
132 
133  // reactivate button's window
134  activateWindow();
135  }

251 void QgsSymbolButton::mouseMoveEvent( QMouseEvent *e )
252 {
253  //handle dragging colors/symbols from button
254 
255  if ( !( e->buttons() & Qt::LeftButton ) )
256  {
257  //left button not depressed, so not a drag
258  QToolButton::mouseMoveEvent( e );
259  return;
260  }
261 
262  if ( ( e->pos() - mDragStartPosition ).manhattanLength() < QApplication::startDragDistance() )
263  {
264  //mouse not moved, so not a drag
265  QToolButton::mouseMoveEvent( e );

271  drag->setMimeData( QgsSymbolLayerUtils::colorToMimeData( mSymbol->color() ) );
272  drag->setPixmap( QgsColorWidget::createDragIcon( mSymbol->color() ) );
273  drag->exec( Qt::CopyAction );
274  setDown( false );

318 void QgsSymbolButton::prepareMenu()
319 {
320  //we need to tear down and rebuild this menu every time it is shown. Otherwise the space allocated to any
321  //QgsColorSwatchGridAction is not recalculated by Qt and the swatch grid may not be the correct size
322  //for the number of colors shown in the grid. Note that we MUST refresh color swatch grids every time this
323  //menu is opened, otherwise color schemes like the recent color scheme grid are meaningless
324  mMenu->clear();
325 
326  QAction *configureAction = new QAction( tr( "Configure Symbol" ), this );
327  mMenu->addAction( configureAction );
328  connect( configureAction, &QAction::triggered, this, &QgsSymbolButton::showSettingsDialog );
329 
330  QAction *copySymbolAction = new QAction( tr( "Copy Symbol" ), this );
331  mMenu->addAction( copySymbolAction );
332  connect( copySymbolAction, &QAction::triggered, this, &QgsSymbolButton::copySymbol );
333  QAction *pasteSymbolAction = new QAction( tr( "Paste Symbol" ), this );
334  //enable or disable paste action based on current clipboard contents. We always show the paste
335  //action, even if it's disabled, to give hint to the user that pasting symbols is possible
336  std::unique_ptr< QgsSymbol > tempSymbol( QgsSymbolLayerUtils::symbolFromMimeData( QApplication::clipboard()->mimeData() ) );
337  if ( tempSymbol && tempSymbol->type() == mType )
338  {
339  const int iconSize = QgsGuiUtils::scaleIconSize( 16 );
340  pasteSymbolAction->setIcon( QgsSymbolLayerUtils::symbolPreviewIcon( tempSymbol.get(), QSize( iconSize, iconSize ), 1 ) );
341  }
342  else
343  {
344  pasteSymbolAction->setEnabled( false );
345  }
346  mMenu->addAction( pasteSymbolAction );
347  connect( pasteSymbolAction, &QAction::triggered, this, &QgsSymbolButton::pasteSymbol );
348 
349  mMenu->addSeparator();
350 
351  QgsColorWheel *colorWheel = new QgsColorWheel( mMenu );
352  colorWheel->setColor( mSymbol->color() );
353  QgsColorWidgetAction *colorAction = new QgsColorWidgetAction( colorWheel, mMenu, mMenu );
354  colorAction->setDismissOnColorSelection( false );
355  connect( colorAction, &QgsColorWidgetAction::colorChanged, this, &QgsSymbolButton::setColor );
356  mMenu->addAction( colorAction );
357 
358  QgsColorRampWidget *alphaRamp = new QgsColorRampWidget( mMenu, QgsColorWidget::Alpha, QgsColorRampWidget::Horizontal );
359  QColor alphaColor = mSymbol->color();
360  alphaColor.setAlphaF( mSymbol->opacity() );
361  alphaRamp->setColor( alphaColor );
362  QgsColorWidgetAction *alphaAction = new QgsColorWidgetAction( alphaRamp, mMenu, mMenu );
363  alphaAction->setDismissOnColorSelection( false );
364  connect( alphaAction, &QgsColorWidgetAction::colorChanged, this, [ = ]( const QColor & color )
365  {
366  double opacity = color.alphaF();
367  mSymbol->setOpacity( opacity );
368  updatePreview();
369  emit changed();
370  } );
371  connect( colorAction, &QgsColorWidgetAction::colorChanged, alphaRamp, [alphaRamp]( const QColor & color ) { alphaRamp->setColor( color, false ); }
372  );
373  mMenu->addAction( alphaAction );
374 
375  //get schemes with ShowInColorButtonMenu flag set
376  QList< QgsColorScheme * > schemeList = QgsApplication::colorSchemeRegistry()->schemes( QgsColorScheme::ShowInColorButtonMenu );
377  QList< QgsColorScheme * >::iterator it = schemeList.begin();
378  for ( ; it != schemeList.end(); ++it )
379  {
380  QgsColorSwatchGridAction *colorAction = new QgsColorSwatchGridAction( *it, mMenu, QStringLiteral( "symbology" ), this );
381  colorAction->setBaseColor( mSymbol->color() );
382  mMenu->addAction( colorAction );
383  connect( colorAction, &QgsColorSwatchGridAction::colorChanged, this, &QgsSymbolButton::setColor );
384  connect( colorAction, &QgsColorSwatchGridAction::colorChanged, this, &QgsSymbolButton::addRecentColor );
385  }
386 
387  mMenu->addSeparator();
388 
389  QAction *copyColorAction = new QAction( tr( "Copy Color" ), this );
390  mMenu->addAction( copyColorAction );
391  connect( copyColorAction, &QAction::triggered, this, &QgsSymbolButton::copyColor );
392 
393  QAction *pasteColorAction = new QAction( tr( "Paste Color" ), this );
394  //enable or disable paste action based on current clipboard contents. We always show the paste
395  //action, even if it's disabled, to give hint to the user that pasting colors is possible
396  QColor clipColor;
397  bool hasAlpha = false;
398  if ( colorFromMimeData( QApplication::clipboard()->mimeData(), clipColor, hasAlpha ) )
399  {
400  pasteColorAction->setIcon( createColorIcon( clipColor ) );
401  }
402  else
403  {
404  pasteColorAction->setEnabled( false );
405  }
406  mMenu->addAction( pasteColorAction );
407  connect( pasteColorAction, &QAction::triggered, this, &QgsSymbolButton::pasteColor );
408 }

439 void QgsSymbolButton::updatePreview( const QColor &color, QgsSymbol *tempSymbol )
440 {
441  std::unique_ptr< QgsSymbol > previewSymbol;
442 
443  if ( tempSymbol )
444  previewSymbol.reset( tempSymbol->clone() );
445  else
446  previewSymbol.reset( mSymbol->clone() );
447 
448  if ( color.isValid() )
449  previewSymbol->setColor( color );
450 
451  QSize currentIconSize;
452  //icon size is button size with a small margin
453  if ( menu() )
454  {
455  if ( !mIconSize.isValid() )
456  {
457  //calculate size of push button part of widget (ie, without the menu dropdown button part)
458  QStyleOptionToolButton opt;
459  initStyleOption( &opt );
460  QRect buttonSize = QApplication::style()->subControlRect( QStyle::CC_ToolButton, &opt, QStyle::SC_ToolButton,
461  this );
462  //make sure height of icon looks good under different platforms
463 #ifdef Q_OS_WIN
464  mIconSize = QSize( buttonSize.width() - 10, height() - 6 );
465 #else
466  mIconSize = QSize( buttonSize.width() - 10, height() - 12 );
467 #endif
468  }
469  currentIconSize = mIconSize;
470  }
471  else
472  {
473  //no menu
474 #ifdef Q_OS_WIN
475  currentIconSize = QSize( width() - 10, height() - 6 );
476 #else
477  currentIconSize = QSize( width() - 10, height() - 12 );
478 #endif
479  }
480 
481  if ( !currentIconSize.isValid() || currentIconSize.width() <= 0 || currentIconSize.height() <= 0 )
482  {
483  return;
484  }
485 
486  //create an icon pixmap
487  QIcon icon = QgsSymbolLayerUtils::symbolPreviewIcon( previewSymbol.get(), currentIconSize );
488  setIconSize( currentIconSize );
489  setIcon( icon );
490 
491  // set tooltip
492  // create very large preview image
493  int width = static_cast< int >( Qgis::UI_SCALE_FACTOR * fontMetrics().width( 'X' ) * 23 );
494  int height = static_cast< int >( width / 1.61803398875 ); // golden ratio
495 
496  QPixmap pm = QgsSymbolLayerUtils::symbolPreviewPixmap( previewSymbol.get(), QSize( width, height ), height / 20 );
497  QByteArray data;
498  QBuffer buffer( &data );
499  pm.save( &buffer, "PNG", 100 );
500  setToolTip( QStringLiteral( "<img src='data:image/png;base64, %3'>" ).arg( QString( data.toBase64() ) ) );
501 }
