100 void QgsLayoutItemLabel::draw( QgsLayoutItemRenderContext &context )
101 {
102  QPainter *painter = context.renderContext().painter();
103  painter->save();
104 
105  // painter is scaled to dots, so scale back to layout units
106  painter->scale( context.renderContext().scaleFactor(), context.renderContext().scaleFactor() );
107 
108  double penWidth = frameEnabled() ? ( pen().widthF() / 2.0 ) : 0;
109  double xPenAdjust = mMarginX < 0 ? -penWidth : penWidth;
110  double yPenAdjust = mMarginY < 0 ? -penWidth : penWidth;
111  QRectF painterRect( xPenAdjust + mMarginX, yPenAdjust + mMarginY, rect().width() - 2 * xPenAdjust - 2 * mMarginX, rect().height() - 2 * yPenAdjust - 2 * mMarginY );
112 
113  switch ( mMode )
114  {
115  case ModeHtml:
116  {
117  if ( mFirstRender )
118  {
119  contentChanged();
120  mFirstRender = false;
121  }
122  painter->scale( 1.0 / mHtmlUnitsToLayoutUnits / 10.0, 1.0 / mHtmlUnitsToLayoutUnits / 10.0 );
123  mWebPage->setViewportSize( QSize( painterRect.width() * mHtmlUnitsToLayoutUnits * 10.0, painterRect.height() * mHtmlUnitsToLayoutUnits * 10.0 ) );
124  mWebPage->settings()->setUserStyleSheetUrl( createStylesheetUrl() );
125  mWebPage->mainFrame()->render( painter );
126  break;
127  }
128 
129  case ModeFont:
130  {
131  const QString textToDraw = currentText();
132  painter->setFont( mFont );
133  QgsLayoutUtils::drawText( painter, painterRect, textToDraw, mFont, mFontColor, mHAlignment, mVAlignment, Qt::TextWordWrap );
134  break;
135  }
136  }
137 
138  painter->restore();
139 }

141 void QgsLayoutItemLabel::contentChanged()
142 {
143  switch ( mMode )
144  {
145  case ModeHtml:
146  {
147  const QString textToDraw = currentText();
148 
149  //mHtmlLoaded tracks whether the QWebPage has completed loading
150  //its html contents, set it initially to false. The loadingHtmlFinished slot will
151  //set this to true after html is loaded.
152  mHtmlLoaded = false;
153 
154  const QUrl baseUrl = QUrl::fromLocalFile( mLayout->project()->absoluteFilePath() );
155  mWebPage->mainFrame()->setHtml( textToDraw, baseUrl );
156 
157  //For very basic html labels with no external assets, the html load will already be
158  //complete before we even get a chance to start the QEventLoop. Make sure we check
159  //this before starting the loop
160  if ( !mHtmlLoaded )
161  {
162  //Setup event loop and timeout for rendering html
163  QEventLoop loop;
164 
165  //Connect timeout and webpage loadFinished signals to loop
166  connect( mWebPage.get(), &QWebPage::loadFinished, &loop, &QEventLoop::quit );
167 
168  // Start a 20 second timeout in case html loading will never complete
169  QTimer timeoutTimer;
170  timeoutTimer.setSingleShot( true );
171  connect( &timeoutTimer, &QTimer::timeout, &loop, &QEventLoop::quit );
172  timeoutTimer.start( 20000 );
173 
174  // Pause until html is loaded
175  loop.exec( QEventLoop::ExcludeUserInputEvents );
176  }
177  break;
178  }
179  case ModeFont:
180  break;
181  }
182 }

266 void QgsLayoutItemLabel::replaceDateText( QString &text ) const
267 {
268  QString constant = QStringLiteral( "$CURRENT_DATE" );
269  int currentDatePos = text.indexOf( constant );
270  if ( currentDatePos != -1 )
271  {
272  //check if there is a bracket just after $CURRENT_DATE
273  QString formatText;
274  int openingBracketPos = text.indexOf( '(', currentDatePos );
275  int closingBracketPos = text.indexOf( ')', openingBracketPos + 1 );
276  if ( openingBracketPos != -1 &&
277  closingBracketPos != -1 &&
278  ( closingBracketPos - openingBracketPos ) > 1 &&
279  openingBracketPos == currentDatePos + constant.size() )
280  {
281  formatText = text.mid( openingBracketPos + 1, closingBracketPos - openingBracketPos - 1 );
282  text.replace( currentDatePos, closingBracketPos - currentDatePos + 1, QDate::currentDate().toString( formatText ) );
283  }
284  else //no bracket
285  {
286  text.replace( QLatin1String( "$CURRENT_DATE" ), QDate::currentDate().toString() );
287  }
288  }
289 }

371 bool QgsLayoutItemLabel::readPropertiesFromElement( const QDomElement &itemElem, const QDomDocument &, const QgsReadWriteContext & )
372 {
373  //restore label specific properties
374 
375  //text
376  mText = itemElem.attribute( QStringLiteral( "labelText" ) );
377 
378  //html state
379  mMode = static_cast< Mode >( itemElem.attribute( QStringLiteral( "htmlState" ) ).toInt() );
380 
381  //margin
382  bool marginXOk = false;
383  bool marginYOk = false;
384  mMarginX = itemElem.attribute( QStringLiteral( "marginX" ) ).toDouble( &marginXOk );
385  mMarginY = itemElem.attribute( QStringLiteral( "marginY" ) ).toDouble( &marginYOk );
386  if ( !marginXOk || !marginYOk )
387  {
388  //upgrade old projects where margins where stored in a single attribute
389  double margin = itemElem.attribute( QStringLiteral( "margin" ), QStringLiteral( "1.0" ) ).toDouble();
390  mMarginX = margin;
391  mMarginY = margin;
392  }
393 
394  //Horizontal alignment
395  mHAlignment = static_cast< Qt::AlignmentFlag >( itemElem.attribute( QStringLiteral( "halign" ) ).toInt() );
396 
397  //Vertical alignment
398  mVAlignment = static_cast< Qt::AlignmentFlag >( itemElem.attribute( QStringLiteral( "valign" ) ).toInt() );
399 
400  //font
401  QgsFontUtils::setFromXmlChildNode( mFont, itemElem, QStringLiteral( "LabelFont" ) );
402 
403  //font color
404  QDomNodeList fontColorList = itemElem.elementsByTagName( QStringLiteral( "FontColor" ) );
405  if ( !fontColorList.isEmpty() )
406  {
407  QDomElement fontColorElem = fontColorList.at( 0 ).toElement();
408  int red = fontColorElem.attribute( QStringLiteral( "red" ), QStringLiteral( "0" ) ).toInt();
409  int green = fontColorElem.attribute( QStringLiteral( "green" ), QStringLiteral( "0" ) ).toInt();
410  int blue = fontColorElem.attribute( QStringLiteral( "blue" ), QStringLiteral( "0" ) ).toInt();
411  mFontColor = QColor( red, green, blue );
412  }
413  else
414  {
415  mFontColor = QColor( 0, 0, 0 );
416  }
417 
418  return true;
419 }

491 void QgsLayoutItemLabel::itemShiftAdjustSize( double newWidth, double newHeight, double &xShift, double &yShift ) const
492 {
493  //keep alignment point constant
494  double currentWidth = rect().width();
495  double currentHeight = rect().height();
496  xShift = 0;
497  yShift = 0;
498 
499  double r = rotation();
500  if ( r >= 0 && r < 90 )
501  {
502  if ( mHAlignment == Qt::AlignHCenter )
503  {
504  xShift = - ( newWidth - currentWidth ) / 2.0;
505  }
506  else if ( mHAlignment == Qt::AlignRight )
507  {
508  xShift = - ( newWidth - currentWidth );
509  }
510  if ( mVAlignment == Qt::AlignVCenter )
511  {
512  yShift = -( newHeight - currentHeight ) / 2.0;
513  }
514  else if ( mVAlignment == Qt::AlignBottom )
515  {
516  yShift = - ( newHeight - currentHeight );
517  }
518  }
519  if ( r >= 90 && r < 180 )
520  {
521  if ( mHAlignment == Qt::AlignHCenter )
522  {
523  yShift = -( newHeight - currentHeight ) / 2.0;
524  }
525  else if ( mHAlignment == Qt::AlignRight )
526  {
527  yShift = -( newHeight - currentHeight );
528  }
529  if ( mVAlignment == Qt::AlignTop )
530  {
531  xShift = -( newWidth - currentWidth );
532  }
533  else if ( mVAlignment == Qt::AlignVCenter )
534  {
535  xShift = -( newWidth - currentWidth / 2.0 );
536  }
537  }
538  else if ( r >= 180 && r < 270 )
539  {
540  if ( mHAlignment == Qt::AlignHCenter )
541  {
542  xShift = -( newWidth - currentWidth ) / 2.0;
543  }
544  else if ( mHAlignment == Qt::AlignLeft )
545  {
546  xShift = -( newWidth - currentWidth );
547  }
548  if ( mVAlignment == Qt::AlignVCenter )
549  {
550  yShift = ( newHeight - currentHeight ) / 2.0;
551  }
552  else if ( mVAlignment == Qt::AlignTop )
553  {
554  yShift = ( newHeight - currentHeight );
555  }
556  }
557  else if ( r >= 270 && r < 360 )
558  {
559  if ( mHAlignment == Qt::AlignHCenter )
560  {
561  yShift = -( newHeight - currentHeight ) / 2.0;
562  }
563  else if ( mHAlignment == Qt::AlignLeft )
564  {
565  yShift = -( newHeight - currentHeight );
566  }
567  if ( mVAlignment == Qt::AlignBottom )
568  {
569  xShift = -( newWidth - currentWidth );
570  }
571  else if ( mVAlignment == Qt::AlignVCenter )
572  {
573  xShift = -( newWidth - currentWidth / 2.0 );
574  }
575  }
576 }
