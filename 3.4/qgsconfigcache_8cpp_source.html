39 const QgsProject *QgsConfigCache::project( const QString &path )
40 {
41  if ( ! mProjectCache[ path ] )
42  {
43  std::unique_ptr<QgsProject> prj( new QgsProject() );
44  if ( prj->read( path ) )
45  {
46  mProjectCache.insert( path, prj.release() );

48  }
49  else

51  QgsMessageLog::logMessage(

53  QStringLiteral( "Server" ), Qgis::Critical );

55  }
56  QgsProject::setInstance( mProjectCache[ path ] );

60 QDomDocument *QgsConfigCache::xmlDocument( const QString &filePath )
61 {
62  //first open file
63  QFile configFile( filePath );
64  if ( !configFile.exists() )
65  {
66  QgsMessageLog::logMessage( "Error, configuration file '" + filePath + "' does not exist", QStringLiteral( "Server" ), Qgis::Critical );
67  return nullptr;
68  }
69 
70  if ( !configFile.open( QIODevice::ReadOnly ) )
71  {
72  QgsMessageLog::logMessage( "Error, cannot open configuration file '" + filePath + "'", QStringLiteral( "Server" ), Qgis::Critical );
73  return nullptr;
74  }
75 
76  // first get cache
77  QDomDocument *xmlDoc = mXmlDocumentCache.object( filePath );
78  if ( !xmlDoc )
79  {
80  //then create xml document
81  xmlDoc = new QDomDocument();
82  QString errorMsg;
83  int line, column;
84  if ( !xmlDoc->setContent( &configFile, true, &errorMsg, &line, &column ) )
85  {
86  QgsMessageLog::logMessage( "Error parsing file '" + filePath +
87  QStringLiteral( "': parse error %1 at row %2, column %3" ).arg( errorMsg ).arg( line ).arg( column ), QStringLiteral( "Server" ), Qgis::Critical );
88  delete xmlDoc;
89  return nullptr;
90  }
91  mXmlDocumentCache.insert( filePath, xmlDoc );
92  mFileSystemWatcher.addPath( filePath );
93  xmlDoc = mXmlDocumentCache.object( filePath );
94  Q_ASSERT( xmlDoc );
95  }
96  return xmlDoc;
97 }
