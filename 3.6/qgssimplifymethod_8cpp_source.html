41  // returns a geometry simplifier according to specified method
42  if ( methodType == QgsSimplifyMethod::OptimizeForRendering )

44  int simplifyFlags = QgsMapToPixelSimplifier::SimplifyGeometry | QgsMapToPixelSimplifier::SimplifyEnvelope;
